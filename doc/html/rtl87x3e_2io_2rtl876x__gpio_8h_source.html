740 __forceinline void GPIOx_Debounce_Time(GPIO_TypeDef *GPIOx, uint32_t DebounceTime)
741 {
742  uint8_t count = 0;
743 
744  if (DebounceTime < 1)
745  {
746  DebounceTime = 1;
747  }
748  if (DebounceTime > 64)
749  {
750  DebounceTime = 64;
751  }
752 #ifdef _IS_ASIC_
753  //div = 14;//0xd 0b1101<<8
754  if (GPIOx == GPIOA)
755  {
756  GPIO_DBCLK_DIV = ((0xd << 8) | (1 << 12));
757  }
758  else if (GPIOx == GPIOB)
759  {
760  GPIO_DBCLK_DIV = ((0xd << 24) | (1 << 28));
761  }
762 
763 #else
764  //div = 13;//0xc 0b1100<<8
765  if (GPIOx == GPIOA)
766  {
767  GPIO_DBCLK_DIV |= ((1 << 11) | (1 << 10) | (1 << 12));
768  GPIO_DBCLK_DIV &= (~((1 << 9) | (1 << 8)));
769  }
770  else if (GPIOx == GPIOB)
771  {
772  GPIO_DBCLK_DIV |= ((1 << 27) | (1 << 26) | (1 << 28));
773  GPIO_DBCLK_DIV &= (~((1 << 25) | (1 << 24)));
774  }
775 #endif
776 
777 
778  count = (244 * DebounceTime) / 100 - 1;
779  if (GPIOx == GPIOA)
780  {
781  GPIO_DBCLK_DIV &= (~((0xff << 0)));
782  GPIO_DBCLK_DIV |= GPIO_DBCLK_DIV + count;
783  }
784  else if (GPIOx == GPIOB)
785  {
786  GPIO_DBCLK_DIV &= (~((0xff << 16)));
787  GPIO_DBCLK_DIV |= (GPIO_DBCLK_DIV + count) << 16;
788  }
