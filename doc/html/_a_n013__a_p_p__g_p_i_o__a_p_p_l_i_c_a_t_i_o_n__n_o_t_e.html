2 GPIO Api Usage

The purpose of this chapter is to give an overview of gpio api. The source code is in sdk\src\sample\rws\app_gpio.c.

2.1 GPIO_Initialization

User can use the following api to Initialize specified gpio pin.

GPIO_InitTypeDef gpio_param; /* Define GPIO parameter structure */
app_gpio_periph_clk_cfg(pin_num, (FunctionalState)ENABLE);
GPIO_StructInit(&gpio_param);
gpio_param.GPIO_PinBit = GPIO_GetPin(pin_num);
gpio_param.GPIO_ITCmd = ENABLE;
gpio_param.GPIO_ITTrigger = int_trig_type;
if (debounce != 0)
{
gpio_param.GPIO_ITDebounce = GPIO_INT_DEBOUNCE_ENABLE;
gpio_param.GPIO_DebounceTime = debounce;
}
else
{
gpio_param.GPIO_ITDebounce = GPIO_INT_DEBOUNCE_DISABLE;
}
app_gpio_param_cfg(pin_num, &gpio_param);
/*disable int*/
app_gpio_int_cfg(pin_num, DISABLE);
app_gpio_mask_int_cfg(pin_num, ENABLE);
app_gpio_clear_int_pending(pin_num);

2.2 GPIO_Parameter_Initialization

User can use the following api to initializes the GPIO A and B peripheral according to the specified parameters in the GPIO_InitStruct.

ISR_TEXT_SECTION void app_gpio_param_cfg(uint8_t pin_num,
GPIO_InitTypeDef *init_struct)
{
if (GPIO_GetNum(pin_num) <= GPIO31)
{
GPIOx_Init(GPIOA, init_struct);
}
#ifndef TARGET_RTL8753GFE
else
{
GPIOx_Init(GPIOB, init_struct);
}
#endif
}

2.3 GPIO_Clock_Configuration

User can use the following api to set specified GPIO APB peripheral clock.

ISR_TEXT_SECTION void app_gpio_periph_clk_cfg(uint8_t pin_num, FunctionalState state)
{
if (GPIO_GetNum(pin_num) <= GPIO31)
{
RCC_PeriphClockCmd(APBPeriph_GPIOA, APBPeriph_GPIOA_CLOCK, state);
}
#ifndef TARGET_RTL8753GFE
else
{
RCC_PeriphClockCmd(APBPeriph_GPIOB, APBPeriph_GPIOB_CLOCK, state);
}
#endif
}

2.7 GPIO_Interrupt_Polarity_Configuartion

User can use the following api to change GPIO change interrupt polarity.

ISR_TEXT_SECTION void app_gpio_int_polarity_cfg(uint8_t pin_num,
GPIOIT_PolarityType polarity)
{
if (GPIO_GetNum(pin_num) <= GPIO31)
{
if (polarity)
{
GPIOA->INTPOLARITY |= GPIO_GetPin(pin_num);
}
else
{
GPIOA->INTPOLARITY &= ~(GPIO_GetPin(pin_num));
}
}
#ifndef TARGET_RTL8753GFE
else
{
if (polarity)
{
GPIOB->INTPOLARITY |= GPIO_GetPin(pin_num);
}
else
{
GPIOB->INTPOLARITY &= ~(GPIO_GetPin(pin_num));
}
}
#endif
}

2.8 GPIO_Interrupt_Enable

User can use the following api to enable Specified GPIO interrupt.

void app_gpio_int_enable(uint8_t pin_num)
{
app_gpio_mask_int_cfg(pin_num, DISABLE);
app_gpio_int_cfg(pin_num, ENABLE);
}

2.9 GPIO_Interrupt_Disable

User can use the following api to disable Specified GPIO interrupt.

void app_gpio_int_disable(uint8_t pin_num)
{
app_gpio_int_cfg(pin_num, DISABLE);
app_gpio_mask_int_cfg(pin_num, ENABLE);
app_gpio_clear_int_pending(pin_num);
}

2.10 GPIO_Pinmux_Level_Configuration

User can use the following api to set the GPIO pinmux level.

void app_gpio_def_level_cfg(uint8_t pin_num, PAD_Pull_Mode level)
{
Pinmux_Config(pin_num, DWGPIO);
if (level == PAD_PULL_UP)
{
Pad_Config(pin_num, PAD_PINMUX_MODE, PAD_IS_PWRON, PAD_PULL_UP, PAD_OUT_DISABLE, PAD_OUT_HIGH);
}
else if (level == PAD_PULL_DOWN)
{
Pad_Config(pin_num, PAD_PINMUX_MODE, PAD_IS_PWRON, PAD_PULL_DOWN, PAD_OUT_DISABLE, PAD_OUT_LOW);
}
}

2.11 GPIO_Pinmux_Output_Level_Set

User can use the following api to set the GPIO pinmux output level.

RAM_TEXT_SECTION void app_gpio_write_output_level(uint8_t pin_num, uint8_t level)
{
if (GPIO_GetNum(pin_num) <= GPIO31)
{
GPIOx_WriteBit(GPIOA, GPIO_GetPin(pin_num), (BitAction)level);

#ifndef TARGET_RTL8753GFE
else
{
GPIOx_WriteBit(GPIOB, GPIO_GetPin(pin_num), (BitAction)level);
}
#endif
}

2.12 GPIO_Pinmux_Input_Level_Read

User can use the following api to read the GPIO pinmux input level.

ISR_TEXT_SECTION uint8_t app_gpio_read_input_level(uint8_t pin_num)
{
uint8_t bit_value = 0;
if (GPIO_GetNum(pin_num) <= GPIO31)
{
bit_value = GPIOx_ReadInputDataBit(GPIOA, GPIO_GetPin(pin_num));
}
#ifndef TARGET_RTL8753GFE
else
{
bit_value = GPIOx_ReadInputDataBit(GPIOB, GPIO_GetPin(pin_num));
}
#endif
return bit_value;
}
