1 Introduction

The purpose of this document is to give an overview of pbap function.

want to use the pbap, should first:
1- #define F_APP_BT_PROFILE_PBAP_SUPPORT 1 in project
2- mcu config tool config as shown in Figure 1-1: app_cfg_const.supported_profile_mask & PBAP_PROFILE_MASK
Figure 1-1 PBAP configuration in mcu config tool

The pbap profile(It co-work with HFP) usually used for tts, two state variable is important:
    p_link->call_id_type_check: if need to do call id type check(number type or name type)
    p_link->call_id_type_num: true mean number type, false mean name type

a. When HFP connected or call idle, p_link->call_id_type_check set to true
b. When pbap profile connected, bud send cmd to source to set phone book repository to local
c. When HFP caller id indication(about once per 3 second), would judge call_id_type_check,
    if true, would do type judge(maybe start source PBAP name inquiry)
    if false, would start tts text(caller number) report to source
d. When PBAP caller name indication, also would start tts text(caller name) report to source

2 Source Code Overview

This section discribes the main parts of pbap source code.

HFP Event handle

static void app_hfp_bt_cback(T_BT_EVENT event_type, void *event_buf, uint16_t buf_len)
{
......
switch (event_type)
{
case BT_EVENT_HFP_CONN_CMPL:
{
if (p_link != NULL)
{
p_link->call_id_type_check = true;
p_link->call_id_type_num = false;
}
}
break;
case BT_EVENT_HFP_CALL_STATUS:
{
if (p_link->call_status == APP_CALL_IDLE)
{
p_link->call_id_type_check = true;
p_link->call_id_type_num = false;
}
}
break;
case BT_EVENT_HFP_CALLER_ID_IND:
{
if (br_link->call_id_type_check == true)
{
if (br_link->connected_profile & PBAP_PROFILE_MASK)
{
//bud inquiry source caller name
if (bt_pbap_vcard_listing_by_number_pull(br_link->bd_addr, number) == false)
{
br_link->call_id_type_check = false;
br_link->call_id_type_num = true;
}
}
else
{
br_link->call_id_type_check = false;
br_link->call_id_type_num = true;
}
}
//if caller id type decided, it is number type, start tts text(caller number) report to source
if (br_link->call_id_type_check == false)
{
if (br_link->call_id_type_num == true)
{
if (br_link->connected_profile & SPP_PROFILE_MASK)
{
app_report_event(CMD_PATH_SPP, EVENT_CALLER_ID, br_link->id, data, len);
}
else if (br_link->connected_profile & IAP_PROFILE_MASK)
{
app_report_event(CMD_PATH_IAP, EVENT_CALLER_ID, br_link->id, data, len);
}
else if (le_link != NULL)
{
app_report_event(CMD_PATH_LE, EVENT_CALLER_ID, le_link->id, data, len);
}
}
}
}
break;
}
......
}

Pbap Event handle

static void app_pbap_bt_cback(T_BT_EVENT event_type, void *event_buf, uint16_t buf_len)
{
......
switch (event_type)
{
//when pbap profile connected, set phone book repository to local
case BT_EVENT_PBAP_CONN_CMPL:
{
T_APP_BR_LINK *p_link;
p_link = app_link_find_br_link(param->pbap_conn_cmpl.bd_addr);
if (p_link != NULL)
{
bt_pbap_phone_book_set(p_link->bd_addr, BT_PBAP_REPOSITORY_LOCAL, BT_PBAP_PATH_PB);
}
}
break;
case BT_EVENT_PBAP_CALLER_ID_NAME:
{
if (app_cfg_nv.bud_role != REMOTE_SESSION_ROLE_SECONDARY)
{
//start tts text(caller number) report to source
if (br_link->connected_profile & (SPP_PROFILE_MASK | IAP_PROFILE_MASK))
{
app_report_event(CMD_PATH_SPP, EVENT_CALLER_ID, br_link->id, data, len);
}
else if (le_link != NULL)
{
app_report_event(CMD_PATH_LE, EVENT_CALLER_ID, le_link->id, data, len);
}
}
}
break;

}
