818 {
819  /* Check the parameters */

821  assert_param(IS_SPI_DIRECTION_MODE(dir));
822 
823  /* Disable the selected SPI peripheral */
824  SPIx->SSIENR &= ~0x01;

827  SPIx->CTRLR0 &= ~(0x03 << 8);

830  /* Enable the selected SPI peripheral */
