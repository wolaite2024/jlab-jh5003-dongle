85 static inline void reverse_array(uint8_t *array, uint8_t start_idx, uint8_t end_idx)
86 {
87  while (start_idx < end_idx)
88  {
89  uint8_t temp = array[start_idx];
90  array[start_idx] = array[end_idx];
91  array[end_idx] = temp;
92  start_idx += 1;
93  end_idx -= 1;
94  }
