}
}

◆ transmission_stream_write()

bool transmission_stream_write ( T_TRANSMISSION_STREAM  stream,
const uint8_t *  data,
uint32_t  length 
)

Write data to the stream.

Note
It means that user will send data by the stream, maximum length of data that can be sent at a time is limited, user can get it by the API transmission_stream_get_mtu. User can try to use stream write function in two cases:
  1. When user have data to send, user can actively call this function. But when write fail, suspending of actively write is recommended.
  2. When the callback function registerd by the user receives the TRANSMISSION_STREAM_READY_TO_WRITE_EVENT, it means that the state of the stream has changed from being unable to send data to being able to send data, user can call this function to continue sending data
Parameters
[in]streamtransmission stream pointer
[in]datathe data you want write to the stream
[in]lengthlength of the data you want to write to the stream
Returns
true when write success, false otherwise

Example Usage

bool send_data(T_TRANSMISSION_STREAM stream, uint8_t *data, uint32_t length)
{
uint32_t remaining_length = length;
uint32_t mtu = transmission_stream_get_mtu(stream);
uint32_t transmission_length = 0;
while(remaining_length != 0)
{
// Calculate the length of data that can be sent at a time
if(remaining_length >= mtu)
{
transmission_length = mtu;
}
else
{
transmission_length = remaining_length;
}
if(transmission_stream_write(stream, data, transmission_length) == true)
{
remaining_length -= transmission_length;
}
else
{
return false;
}
}
return true;
}

◆ transmission_stream_read()

bool transmission_stream_read ( T_TRANSMISSION_STREAM  stream,
uint32_t  length,
uint8_t *  buffer 
)

Read data in the stream.

Note
When you receive the event TRANSMISSION_STREAM_READY_TO_READ_EVENT, it means that stream have reviced data, user can call this function to read the data from stream.User can call the API transmission_stream_get_bytes_available to get the length of data in the stream
Parameters
[in]streamtransmission stream pointer
[in]lengththe length you want to read from the stream
[out]bufferthe data that read from the stream.
Returns
true when read success, false otherwise

Example Usage

T_TRANSMISSION_STREAM stream;
// ...
// Initialize stream and wait to stream is connected
// ...
void stream_ready_to_read_event_process(void)
{
uint32_t length = transmission_stream_get_bytes_available(stream);
uint8_t *buffer = os_mem_alloc(RAM_TYPE_BUFFER_OFF, length);
if(buffer == NULL)
{
return;
}
if(transmission_stream_read(stream, length, buffer) == false)
{
os_mem_free(buffer);
return;
}
// ...
// Processing of received data
// ...
}
void stream_callback(T_TRANSMISSION_STREAM_EVENT event)
{
switch (event)
{
case TRANSMISSION_STREAM_READY_TO_READ_EVENT:
{
stream_ready_to_read_event_process();
}
break;
// ...
default:
break;
}
}

◆ transmission_stream_set_instance_for_register_callback()

bool transmission_stream_set_instance_for_register_callback ( T_TRANSMISSION_STREAM  stream,
P_STREAM_REGISTER_CALLBACK  register_callback 
)

Set instance for stream register callback function.

Note
Register callback is a function pointer in the transmission stream, sub-module that instancing abstract transmission stream should implement register callback function. With this API, sub-module can assign the implemented function to the funtion pointer to instance abstract interface. This API is for the sub-module, not for the user to operate the stream. it is recommended for use in sub-module initialization functions. The return value of the initialization function need to be T_TRANSMISSION_STREAM type
Parameters
[in]streamtransmission stream pointer
[in]register_callbackimplemented register callback function
Returns
true when read success, false otherwise

Example Usage

static bool xxx_stream_register_callback(T_TRANSMISSION_STREAM stream, T_TRANSMISSION_STREAM_CB callback)
{
// ...
}
T_TRANSMISSION_STREAM xxx_stream_init(void)
{
T_TRANSMISSION_STREAM stream = transmission_stream_create();
if(stream)
{
if(transmission_stream_set_instance_for_register_callback(stream, xxx_stream_register_callback) == false)
{
return NULL;
}
}
}

◆ transmission_stream_set_instance()

bool transmission_stream_set_instance ( T_TRANSMISSION_STREAM  stream,
void *  stream_instance 
)

Assigne stream implemented by sub-module to transmission stream.

Note
The stream pointer implemented by the sub-module can be stored in the transmission stream, by this API, so when the transmission stream call the API to control the sub-module(for example transmission_stream_get_mtu), the function in sub-module can get the required information from the saved stream pointer. This API is for the sub-module, not for the user to operate the stream.it is recommended for use in sub-module initialization functions. The return value of the initialization function need to be T_TRANSMISSION_STREAM type
Parameters
[in]streamtransmission stream pointer
[in]stream_instancestream instance of the sub-module
Returns
true when read success, false otherwise

Example Usage

T_TRANSMISSION_STREAM xxx_stream_init(void)
{
T_XXX_STREAM xxx_stream_instance = os_mem_alloc(RAM_TYPE_DATA_ON, sizeof(T_XXX_STREAM_STRUCT));
if (xxx_stream_instance)
{
T_TRANSMISSION_STREAM stream = transmission_stream_create();
if (stream)
{
transmission_stream_set_instance(stream, (void *)xxx_stream_instance);
}
}
}
