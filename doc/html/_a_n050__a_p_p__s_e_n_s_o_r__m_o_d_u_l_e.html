4.2 Select Different Sensor Module

User may choose a different sensor module by using the macro. source code directory: sdk/src/sample/rws/app_flags.h

#define F_APP_SENSOR_SUPPORT 1
#define F_APP_SENSOR_KEY_SUPPORT 1
#define F_APP_SENSOR_LD_SUPPORT 1
#define F_APP_SENSOR_SPATIAL_AUDIO_SUPPORT 0
#if F_APP_SENSOR_KEY_SUPPORT
#undef F_APP_SENSOR_KEY_IO_SUPPORT
#define F_APP_SENSOR_KEY_IO_SUPPORT 1
#endif
#if F_APP_SENSOR_LD_SUPPORT
#define F_APP_SENSOR_LD_IO_SUPPORT 1
#define F_APP_SENSOR_HX3001_SUPPORT 1
#define F_APP_SENSOR_JSA1225_SUPPORT 1
#define F_APP_SENSOR_JSA1227_SUPPORT 1
#define F_APP_SENSOR_PX318J_SUPPORT 0
#endif
#if F_APP_SENSOR_LD_SUPPORT||F_APP_SENSOR_KEY_SUPPORT
#define F_APP_SENSOR_SC7A20_SUPPORT 1
#define F_APP_SENSOR_IQS773_SUPPORT 1
#define APP_SENSOR_IQS773_FINE_TUNE_EN 0
#define F_APP_SENSOR_IQS873_SUPPORT 0
#define F_APP_SENSOR_TOUCH_SUPPORT 0
#endif

4.4 Add Custom Sensor Setting

Source code directory: sdk/src/sample/rws/app_main.c
Add i2c and detect pinmux setting in the following form.

static void board_init(void)
{
...
//Config I2C0 pinmux. For CP control
T_I2C_STATUS status = I2C_STATUS_OK;
status = hal_i2c_init_pin(HAL_I2C_ID_0, app_cfg_const.i2c_0_clk_pinmux,
app_cfg_const.i2c_0_dat_pinmux);
...
}

Initialize custom sensor peripheral.

static void driver_init(void)
{
...
#if F_APP_SENSOR_SUPPORT
app_sensor_init();
#endif
...
}

Source code directory: sdk/src/sample/rws/app_sensor.c

void app_sensor_init(void)
{
#if F_APP_SENSOR_KEY_SUPPORT
app_sensor_key_init();
#endif
#if F_APP_SENSOR_LD_SUPPORT
app_sensor_ld_init();
#endif
#if F_APP_SENSOR_SPATIAL_AUDIO_SUPPORT
app_sensor_spatial_init();
#endif
#if F_APP_SENSOR_TOUCH_SUPPORT
app_sensor_touch_init();
#endif
}

4.5 Handle Custom Sensor Message

Add custom_sensor.c in source code directory: sdk/src/sample/rws
Implement custom_sensor related implementation.
To handle custom sensor interrupt and send io msg, you can implement the interrupt callback in the following form.

void custom_sensor_int_gpio_intr_callback(void)
{
...
/* Get detect status */
...
/* send io msg */
gpio_msg.type = IO_MSG_TYPE_GPIO;
gpio_msg.subtype = IO_MSG_GPIO_CUSTOM_SENSOR;
gpio_msg.u.param = custom_status;
if (gpio_send_io_msg(&gpio_msg) == false)
{
APP_PRINT_ERROR0("Send io msg error");
}
...
}

Source code directory: sdk/src/sample/rws/app_io_msg.c
Handle io msg which is sent from custom interrupt handler.

void app_io_handle_msg(T_IO_MSG io_driver_msg_recv)
{
...
switch (msgtype)
{
...
case IO_MSG_GPIO_CUSTOM_SENSOR:
app_custom_sensor_handle_msg(&io_driver_msg_recv);
break;
}
...
}

4.6 Integrate Custom Control Function

To enable the custom sensor, you can refer to following operations(ld sensor as example).

void app_sensor_ld_start(void)
{
...
#if F_APP_SENSOR_SC7A20_SUPPORT
if (app_cfg_const.ld_sensor_vendor == SENSOR_VENDOR_SC7A20)
{
sc7a20_ld_state = SENSOR_LD_NONE;
sensor_sc7a20_enable();
app_sensor_ld_start_poll_timer(SC7A20_LD_POLL_MS);
}
#endif
}
void app_sensor_ld_stop(void)
{
...
#if F_APP_SENSOR_SC7A20_SUPPORT
if (app_cfg_const.ld_sensor_vendor == SENSOR_VENDOR_SC7A20)
{
app_sensor_ld_stop_poll_timer();
sensor_sc7a20_disable();
}
#endif
}
