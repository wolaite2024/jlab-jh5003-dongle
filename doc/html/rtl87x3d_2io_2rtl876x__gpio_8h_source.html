884 __STATIC_ALWAYS_INLINE void GPIOx_Debounce_Time(GPIO_TypeDef *GPIOx, uint32_t DebounceTime)
885 {
886  uint8_t count = 0;
887 
888  if (DebounceTime < 1)
889  {
890  DebounceTime = 1;
891  }
892  if (DebounceTime > 64)
893  {
894  DebounceTime = 64;
895  }
896 #ifdef _IS_ASIC_
897  //div = 14;//0xd 0b1101<<8
898  if (GPIOx == GPIOA)
899  {
900  GPIOAB_DBCLK_DIV = ((0xd << 8) | (1 << 12));
901  }
902  else if (GPIOx == GPIOB)
903  {
904  GPIOAB_DBCLK_DIV = ((0xd << 24) | (1 << 28));
905  }
906  else if (GPIOx == GPIOC)
907  {
908  GPIOC_DBCLK_DIV = ((0xd << 8) | (1 << 12));
909  }
910 
911 #else
912  //div = 13;//0xc 0b1100<<8
913  if (GPIOx == GPIOA)
914  {
915  GPIOAB_DBCLK_DIV |= ((1 << 11) | (1 << 10) | (1 << 12));
916  GPIOAB_DBCLK_DIV &= (~((1 << 9) | (1 << 8)));
917  }
918  else if (GPIOx == GPIOB)
919  {
920  GPIOAB_DBCLK_DIV |= ((1 << 27) | (1 << 26) | (1 << 28));
921  GPIOAB_DBCLK_DIV &= (~((1 << 25) | (1 << 24)));
922  }
923  else if (GPIOx == GPIOC)
924  {
925  GPIOC_DBCLK_DIV |= ((1 << 11) | (1 << 10) | (1 << 12));
926  GPIOC_DBCLK_DIV &= (~((1 << 9) | (1 << 8)));
927  }
928 #endif
929 
930 
931  count = (244 * DebounceTime) / 100 - 1;
932  if (GPIOx == GPIOA)
933  {
934  GPIOAB_DBCLK_DIV &= (~((0xff << 0)));
935  GPIOAB_DBCLK_DIV |= GPIOAB_DBCLK_DIV + count;
936  }
937  else if (GPIOx == GPIOB)
938  {
939  GPIOAB_DBCLK_DIV &= (~((0xff << 16)));
940  GPIOAB_DBCLK_DIV |= (GPIOAB_DBCLK_DIV + count) << 16;
941  }
942  else if (GPIOx == GPIOC)
943  {
